pairs.py — compute stats & drive header colour
# ── 1. compute after OLS fit ──────────────────────────────────────────
import numpy as np

def pair_stats(dates, spread):
    xs   = np.arange(len(spread), dtype=float)
    slope, intercept = np.polyfit(xs, spread, 1)
    trend   = intercept + slope * xs
    resid   = spread - trend
    sigma   = resid.std()
    last_z  = resid[-1] / sigma if sigma > 0 else 0.0
    return {
        'last_bp'  : round(spread[-1], 1),
        'slope'    : round(slope, 3),        # bp/day
        'last_z'   : round(last_z, 2),
        'trend'    : trend,
        'upper'    : trend + sigma,
        'lower'    : trend - sigma,
    }

# ── 2. colour helper ──────────────────────────────────────────────────
def z_colour(z):
    """Returns hex colour for a z-score value."""
    az = abs(z)
    if   az >= 2.0: return '#e06060'   # red   — signal
    elif az >= 1.5: return '#e8a13f'   # amber — watch
    else:          return '#a4b6d2'   # muted — neutral

# ── 3. build the card header (Dash html.Div) ──────────────────────────
from dash import html

def pair_card_header(leg1, leg2, stats):
    z    = stats['last_z']
    sign = '+' if z >= 0 else ''
    slope_sign = '+' if stats['slope'] >= 0 else ''
    return html.Div([
        html.Div([
            html.B(leg1, style={'color': '#e9eef8'}),
            html.Span(' vs ', style={'color': '#4a5d7c'}),
            html.B(leg2, style={'color': '#e9eef8'}),
        ], style={'fontSize': '11px', 'fontWeight': '700',
                  'letterSpacing': '0.09em', 'textTransform': 'uppercase',
                  'color': '#6f83a3'}),
        html.Div([
            html.Span(['last ', html.B(f'{stats["last_bp"]:+.1f} bp')]),
            html.Span(['β ',    html.B(f'{slope_sign}{stats["slope"]:.3f}/d')]),
            html.Span(['z ',
                html.Span(f'{sign}{z:.1f}σ',
                          style={'color': z_colour(z), 'fontWeight': '700'}),
            ]),
        ], style={'display': 'flex', 'gap': '16px',
                  'fontSize': '12px', 'color': '#6f83a3'}),
    ], style={
        'padding': '13px 18px',
        'borderBottom': '1px solid rgba(255,255,255,0.06)',
        'display': 'flex', 'alignItems': 'baseline',
        'justifyContent': 'space-between', 'gap': '12px',
    })
